def calculate_maxpulse(elapsed_time)
  p1, p2 = maxpulse1.to_f, maxpulse2.to_f
  ratio  = p1 / p2
  psecs  = ratio * elapsed_time.secs.to_f
  ptime  = M26::ElapsedTime.new(psecs)
  ptime.as_hhmmss
end

def maxpulse(age)
  p = 220.0 - age
  p = 196.0 if p > 196.0  # age 24 is "peak" (220-24=196)
  p = 1.0   if p < 1
  p
end
