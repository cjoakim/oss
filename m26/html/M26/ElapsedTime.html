def initialize(raw_val)
  @hh, @mm, @ss = 0, 0, 0.0;
  if (raw_val.kind_of? String)
    initialize_string(raw_val)
  else
    initialize_number(raw_val)
  end
end

def subtract(another_instance)
  h1 = as_hours
  h2 = another_instance.as_hours
  h3 = h1 - h2
  ElapsedTime.new(h3 * SECONDS_PER_HOUR)
end
